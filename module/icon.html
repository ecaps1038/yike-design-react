基本使用

从 @yike-design/react-icons 中引入需要的图标，然后直接使用即可

import { Space } from '@yike-design/react';
import { AvatarOutlined, TickCircleOutlined } from '@yike-design/react-icons';

export default () => {
  return (
    <Space>
      <AvatarOutlined />
      <TickCircleOutlined />
    </Space>
  );
};

旋转角度

通过 rotate 属性设置图标旋转角度，单位为度数。

import { Space } from '@yike-design/react';
import { WordOutlined } from '@yike-design/react-icons';

export default () => {
  return (
    <Space>
      <WordOutlined />
      <WordOutlined rotate={45} />
      <WordOutlined rotate={90} />
      <WordOutlined rotate={180} />
    </Space>
  );
};
